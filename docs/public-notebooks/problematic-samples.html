import numpy as np
import plotly.graph_objects as go
from scipy.stats import gaussian_kde

# Calculate the KDE
kde = gaussian_kde(maes)
x_range = np.linspace(min(maes), max(maes), 1000)
y_kde = kde(x_range)

# Create the density plot
fig = go.Figure()

fig.add_trace(
    go.Scatter(x=x_range, y=y_kde, mode="lines", fill="tozeroy", name="MAE Density")
)

# Add vertical lines for the quantiles
q95 = np.quantile(maes, 0.95)
fig.add_vline(
    x=q95,
    line_dash="dash",
    line_color="red",
    annotation_text="95% quantile",
    annotation_position="top right",
)

# Update layout
fig.update_layout(
    title="Density Plot of Mean Absolute Errors (MAEs)",
    xaxis_title="MAE",
    yaxis_title="Density",
    showlegend=False,
)

# Show the plot
fig.show()

import ase
import ase.visualize
import ipywidgets as widgets
from IPython.display import clear_output, display


# Function to display structure
def display_structure(index):
    data = problematic_data_list[index]

    atoms = ase.Atoms(
        numbers=data.atomic_numbers.numpy(),
        positions=data.pos.numpy(),
        cell=data.cell.numpy().squeeze(),
        pbc=[True, True, True],
    )

    return nglview.show_ase(atoms)


# Create widgets
index_slider = widgets.IntSlider(
    value=0,
    min=0,
    max=len(problematic_data_list) - 1,
    step=1,
    description="Sample Index:",
    continuous_update=False,
)

output = widgets.Output()


# Update function
def update_view(change):
    with output:
        clear_output(wait=True)
        display(display_structure(change.new))


# Connect the slider to the update function
index_slider.observe(update_view, names="value")

# Display everything
display(fig)
display(widgets.VBox([index_slider, output]))

    display(display_structure(0))

def _name_postprocess(name: str):
    if name == "s2ef_s2re":
        name = "base"
    name = name.replace(".0", "")
    if name == "posaug01":
        name = "posaug0.01"
    return name


def _sortidx(name: str):
    mapping = {
        "s2ef_s2re": 0,
        "ec20.0_fc20.0_sc10.0": 1,
        "ec100.0_fc100.0_sc1.0": 2,
        "ec1.0_fc100.0_sc1.0": 3,
        "wd0.1": 4,
        "posaug0.01": 5,
    }
    return mapping.get(name, len(mapping) + 1)
